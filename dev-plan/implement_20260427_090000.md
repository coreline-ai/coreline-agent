### 3.1 `src/tui/theme/types.ts`

```typescript
export interface ThemePalette {
  primaryAccent: string;    // 주 강조색
  background: string;
  foreground: string;
  secondaryAccent: string;  // 보조 강조색
  alert: string;            // 에러/경고
  success: string;          // 성공
}

export interface ThemeStyles {
  info: string;
  error: string;
  success: string;
  warning: string;
  tool: string;
  toolDetail: string;
  response: string;
  reasoning: string;
  user: string;
  primary: string;
  secondary: string;
  muted: string;
  border: string;
}

export interface ThemeDefinition {
  id: string;
  name: string;
  description: string;
  isDark: boolean;
  palette: ThemePalette;
}
```
